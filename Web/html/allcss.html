Mọi điều về CSS

Dưới đây là danh sách tất cả các thuộc tính CSS, với mô tả và ví dụ sử dụng.

Thuộc tính Mô tả Ví dụ
--* Thuộc tính tùy chỉnh cho biến CSS --main-color: red;
accent-color Đặt màu nhấn cho các điều khiển biểu mẫu và các phần tử tương tác accent-color: blue;
align-content Căn chỉnh các dòng của container flex khi có không gian thừa trên trục chéo align-content: center;
align-items Căn chỉnh các item flex dọc theo trục chéo align-items: stretch;
align-self Căn chỉnh một item flex dọc theo trục chéo align-self: flex-start;
alignment-baseline Đặt cách một phần tử cấp inline căn chỉnh baseline của nó với dòng alignment-baseline: baseline;
all Đặt lại tất cả các thuộc tính về giá trị ban đầu hoặc kế thừa all: initial;
anchor-name Định nghĩa một anchor được đặt tên cho định vị anchor-name: --my-anchor;
animation Thuộc tính viết tắt cho các thuộc tính animation animation: spin 2s linear infinite;
animation-composition Đặt cách các animation tổng hợp animation-composition: replace;
animation-delay Đặt thời điểm bắt đầu animation animation-delay: 2s;

animation-name Đặt tên của animation @keyframes animation-name: fade;
animation-play-state Đặt xem animation đang chạy hay tạm dừng animation-play-state: paused;
animation-range Định nghĩa phạm vi của animation animation-range: 0% 100%;
animation-range-end Đặt kết thúc của phạm vi animation animation-range-end: 100%;

backdrop-filter Áp dụng hiệu ứng đồ họa cho khu vực phía sau một phần tử backdrop-filter: blur(5px);
backface-visibility Kiểm soát xem một phần tử có hiển thị khi không đối diện với màn hình backface-visibility: hidden;
background Thuộc tính viết tắt cho các thuộc tính background background: #f00 url(image.jpg) no-repeat;
background-attachment Đặt cách hình nền được gắn background-attachment: fixed;
background-blend-mode Đặt cách hình nền của phần tử hòa trộn với màu nền background-blend-mode: multiply;
background-clip Đặt khu vực vẽ của nền background-clip: border-box;
background-color Đặt màu nền của một phần tử background-color: #ffffff;
background-image Đặt hình nền background-image: url(image.jpg);

background-position-y Đặt vị trí y của hình nền background-position-y: top;
background-repeat Đặt cách hình nền lặp lại background-repeat: no-repeat;
background-size Đặt kích thước của hình nền background-size: cover;
block-size Đặt kích thước của phần tử theo hướng block block-size: 200px;

border-block-start-width Đặt chiều rộng của border block-start border-block-start-width: 1px;
border-block-style Đặt kiểu của các border block logic border-block-style: solid;
border-block-width Đặt chiều rộng của các border block logic border-block-width: 1px;
